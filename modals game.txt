<script>
    // Timer Logic
    let time = 35 * 60; // 35 minutes in seconds
    const timerElement = document.getElementById('timer');

    const countdown = setInterval(() => {
        let minutes = Math.floor(time / 60);
        let seconds = time % 60;
        
        seconds = seconds < 10 ? '0' + seconds : seconds;
        timerElement.innerHTML = `${minutes}:${seconds}`;

        if (time <= 0) {
            clearInterval(countdown);
            document.getElementById('game-content').innerHTML = "<h2 style='color:red;'>GAME OVER! لقد انتهى الوقت.</h2>";
        }
        time--;
    }, 1000);

    // Password Check Logic
    function checkPassword() {
        const pass = document.getElementById('finalPass').value.trim().toLowerCase();
        if (pass === "i am so proud of you") {
            clearInterval(countdown);
            alert("CONGRATULATIONS! 🏆\nلقد نجحتم في فك التشفير والهروب!");
            document.querySelector('.game-container').style.borderColor = "#00ff41";
            document.querySelector('.game-container').innerHTML = "<h1 style='color:#00ff41;'>🎉 MISSION ACCOMPLISHED! 🎉</h1><p style='font-size:2rem;'>I am so proud of you</p>";
        } else {
            alert("كلمة السر خاطئة! حاولوا مجدداً.");
        }
    }
</script>
